suppressPackageStartupMessages(library(leaflet))

pick_path <- function(paths) {
  for (p in paths) if (file.exists(p)) return(p)
  ""
}

flows_path <- pick_path(c(
  "../../data/derived/faf_top_od_flows.csv",
  "../data/derived/faf_top_od_flows.csv",
  "data/derived/faf_top_od_flows.csv"
))
zones_path <- pick_path(c(
  "../../data/derived/faf_zone_centroids.csv",
  "../data/derived/faf_zone_centroids.csv",
  "data/derived/faf_zone_centroids.csv"
))

if (!nzchar(flows_path) || !nzchar(zones_path)) {
  cat("<p><em>Flow map data not yet generated. Run <code>tools/derive_ui_artifacts.R</code> before rendering.</em></p>")
  knitr::knit_exit()
}

flows <- if (nzchar(flows_path)) utils::read.csv(flows_path, stringsAsFactors = FALSE) else data.frame()
zones <- if (nzchar(zones_path)) utils::read.csv(zones_path, stringsAsFactors = FALSE) else data.frame()

if (nrow(flows) == 0 || nrow(zones) == 0) {
  cat("<p><em>Flow map data not yet generated. Run <code>tools/derive_ui_artifacts.R</code> before rendering.</em></p>")
  knitr::knit_exit()
}

required_flows <- c("origin_id", "dest_id", "tons", "ton_miles", "distance_miles", "commodity_group", "scenario_id")
required_zones <- c("zone_id", "name", "lat", "lon")
missing_flows <- setdiff(required_flows, names(flows))
missing_zones <- setdiff(required_zones, names(zones))
if (length(missing_flows) > 0) stop("faf_top_od_flows.csv missing required columns: ", paste(missing_flows, collapse = ", "))
if (length(missing_zones) > 0) stop("faf_zone_centroids.csv missing required columns: ", paste(missing_zones, collapse = ", "))

zones$zone_id <- as.character(zones$zone_id)
flows$origin_id <- as.character(flows$origin_id)
flows$dest_id <- as.character(flows$dest_id)

orig <- zones[match(flows$origin_id, zones$zone_id), c("zone_id", "name", "lat", "lon")]
dest <- zones[match(flows$dest_id, zones$zone_id), c("zone_id", "name", "lat", "lon")]
names(orig) <- c("origin_id_ref", "origin_name", "lat1", "lon1")
names(dest) <- c("dest_id_ref", "dest_name", "lat2", "lon2")
edges <- cbind(flows, orig, dest)
for (nm in c("lat1", "lon1", "lat2", "lon2")) edges[[nm]] <- suppressWarnings(as.numeric(edges[[nm]]))
edges <- edges[is.finite(edges$lat1) & is.finite(edges$lon1) & is.finite(edges$lat2) & is.finite(edges$lon2), , drop = FALSE]

if (nrow(edges) == 0) {
  cat("<p><em>No mappable edges found in UI artifacts.</em></p>")
  knitr::knit_exit()
}

rank_desc <- function(x) rank(-x, ties.method = "first")
edges$rank_tons <- ave(edges$tons, edges$scenario_id, FUN = rank_desc)
edges$rank_tmiles <- ave(edges$ton_miles, edges$scenario_id, FUN = rank_desc)
edges$w <- 1 + 6 * (edges$tons / max(edges$tons, na.rm = TRUE))
edges$class_name <- paste0(
  "flow-edge ",
  "scen-", gsub("[^A-Za-z0-9]", "-", toupper(edges$scenario_id)), " ",
  "rt-", edges$rank_tons, " ",
  "rm-", edges$rank_tmiles
)
pal <- colorFactor(c("#1b9e77", "#d95f02"), domain = unique(edges$scenario_id))

cat(
  "<div style='display:flex; gap:16px; align-items:center; margin-bottom:10px;'>",
  "<label>Top N <select id='flow-topn'><option>10</option><option>25</option><option selected>50</option></select></label>",
  "<label>Weight <select id='flow-weight'><option value='tons' selected>tons</option><option value='ton_miles'>ton_miles</option></select></label>",
  "<label>Scenario <select id='flow-scenario'><option value='ALL' selected>ALL</option>",
  paste(sprintf("<option value='%s'>%s</option>", unique(edges$scenario_id), unique(edges$scenario_id)), collapse = ""),
  "</select></label>",
  "</div>",
  sep = ""
)

map <- leaflet() |>
  addProviderTiles("OpenStreetMap.Mapnik")

for (i in seq_len(nrow(edges))) {
  e <- edges[i, , drop = FALSE]
  popup_txt <- sprintf(
    "<b>%s → %s</b><br/>Scenario: %s<br/>Tons: %.2f<br/>Ton-miles: %.2f<br/>Distance: %.1f mi",
    e$origin_name, e$dest_name, e$scenario_id, e$tons, e$ton_miles, e$distance_miles
  )
  map <- map |>
    addPolylines(
      lng = c(e$lon1, e$lon2),
      lat = c(e$lat1, e$lat2),
      color = pal(e$scenario_id),
      weight = e$w,
      opacity = 0.55,
      options = pathOptions(className = e$class_name),
      popup = popup_txt
    )
}

map <- map |>
  addLegend("bottomright", pal = pal, values = unique(edges$scenario_id), title = "Scenario", opacity = 1) |>
  htmlwidgets::onRender(
    "function(el, x) {
      function applyFilters() {
        var topN = parseInt(document.getElementById('flow-topn').value, 10);
        var metric = document.getElementById('flow-weight').value;
        var scen = document.getElementById('flow-scenario').value;
        var paths = el.querySelectorAll('.flow-edge');
        paths.forEach(function(p) {
          // Avoid regex/backslash escaping pitfalls by using DOMTokenList.
          // Fallback to attribute split for older SVG implementations.
          var classes = p.classList ? Array.from(p.classList) : (p.getAttribute('class') || '').trim().split(' ').filter(Boolean);
          var rank = null;
          var scenarioToken = null;
          classes.forEach(function(c) {
            if (c.indexOf('rt-') === 0 && metric === 'tons') rank = parseInt(c.slice(3), 10);
            if (c.indexOf('rm-') === 0 && metric === 'ton_miles') rank = parseInt(c.slice(3), 10);
            if (c.indexOf('scen-') === 0) scenarioToken = c.replace('scen-', '');
          });
          var keepScenario = (scen === 'ALL') || ((scenarioToken || '').toUpperCase() === scen.toUpperCase());
          var keepRank = (rank !== null && rank <= topN);
          var keep = keepScenario && keepRank;
          p.style.opacity = keep ? '0.62' : '0';
          p.style.pointerEvents = keep ? 'auto' : 'none';
        });
      }
      ['flow-topn','flow-weight','flow-scenario'].forEach(function(id) {
        var node = document.getElementById(id);
        if (node) node.addEventListener('change', applyFilters);
      });
      applyFilters();
    }"
  )
