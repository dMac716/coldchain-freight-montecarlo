pick_path <- function(paths) {
  for (p in paths) if (file.exists(p)) return(p)
  ""
}

summary_path <- pick_path(c(
  "../../data/derived/scenario_summary.csv",
  "../data/derived/scenario_summary.csv",
  "data/derived/scenario_summary.csv"
))

if (!nzchar(summary_path)) {
  summary_path <- pick_path(c(
    "../../outputs/aggregate/results_summary.csv",
    "../outputs/aggregate/results_summary.csv",
    "outputs/aggregate/results_summary.csv"
  ))
}

if (!nzchar(summary_path)) {
  cat("<p><em>Scenario summary not yet generated. Run <code>Rscript tools/derive_ui_artifacts.R</code> (or generate <code>outputs/aggregate/results_summary.csv</code>).</em></p>")
  knitr::knit_exit()
}

df <- utils::read.csv(summary_path, stringsAsFactors = FALSE)
if (!("scenario_id" %in% names(df))) df$scenario_id <- "AGGREGATE"
if (nrow(df) == 0) {
  cat("<p><em>Scenario summary is empty.</em></p>")
  knitr::knit_exit()
}
if (!("metric" %in% names(df))) {
  cat("<p><em>Scenario summary is missing required column <code>metric</code>. Regenerate artifacts via <code>Rscript tools/derive_ui_artifacts.R</code>.</em></p>")
  knitr::knit_exit()
}
metric_vals <- trimws(as.character(df$metric))
if (all(!nzchar(metric_vals))) {
  cat("<p><em>Scenario summary has an empty/invalid <code>metric</code> column. Regenerate artifacts via <code>Rscript tools/derive_ui_artifacts.R</code>.</em></p>")
  knitr::knit_exit()
}
if (!("powertrain" %in% names(df))) df$powertrain <- "unknown"
if (!("trailer_type" %in% names(df))) df$trailer_type <- "unknown"
if (!("variant_id" %in% names(df))) df$variant_id <- df$scenario_id

for (nm in c("mean", "var", "p05", "p50", "p95")) {
  if (!(nm %in% names(df))) df[[nm]] <- NA_real_
}
